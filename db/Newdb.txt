CREATE TABLE Users (
Username TEXT NOT NULL PRIMARY KEY,
Password TEXT NOT NULL,
FirstName TEXT NOT NULL,
LastName TEXT NOT NULL,
Email TEXT
);

CREATE TABLE Projects (
PID INTEGER PRIMARY KEY,
PName VARCHAR NOT NULL,
PDescription VARCHAR NOT NULL,
AutoSave INTEGER NOT NULL,
Admin TEXT NOT NULL,
FOREIGN KEY(Admin) REFERENCES Users(Username)
);

CREATE TABLE Access (
Username TEXT NOT NULL,
PID INTEGER NOT NULL,
FOREIGN KEY(Username) REFERENCES Users(Username),
FOREIGN KEY(PID) REFERENCES Projects(PID)
);

CREATE TABLE Images (
IID INTEGER PRIMARY KEY,
IName VARCHAR NOT NULL,
PID INTEGER NOT NULL,
reviewImage INTEGER NOT NULL DEFAULT 0,
FOREIGN KEY(PID) REFERENCES Projects(PID)
);

CREATE TABLE Classes (
CID INTEGER PRIMARY KEY,
CName VARCHAR NOT NULL,
CStaticID INTEGER NOT NULL,
PID INTEGER NOT NULL,
FOREIGN KEY(PID) REFERENCES Projects(PID)
);

CREATE TABLE Labels (
LID INTEGER PRIMARY KEY,
CID INTEGER NOT NULL,
X INTEGER NOT NULL,
Y INTEGER NOT NULL,
W INTEGER NOT NULL,
H INTEGER NOT NULL,
CStaticID INTEGER NOT NULL,
DateCreated INTEGER NOT NULL,
CreatedBy TEXT NOT NULL,
Type VARCHAR NOT NULL,
IID INTEGER NOT NULL,
FOREIGN KEY(CID) REFERENCES Classes(CID),
FOREIGN KEY(IID) REFERENCES Images(IID)
);
